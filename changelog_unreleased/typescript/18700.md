#### Respect `quoteProps` option when printing enum keys (#18700, #18703 by @fisker)

<!-- prettier-ignore -->
```tsx
// Input
enum E {
  A1 = 0,
  "A2" = 1,
  "A-3" = 1,
}

// Prettier stable
enum E {
  A1 = 0,
  "A2" = 1,
  "A-3" = 1,
}

// Prettier main (--quote-props=as-needed)
enum E {
  A1 = 0,
  A2 = 1,
  "A-3" = 1,
}

// Prettier main (--quote-props=consistent)
enum E {
  "A1" = 0,
  "A2" = 1,
  "A-3" = 1,
}
```
