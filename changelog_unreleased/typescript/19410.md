#### Preserve comment position after `=` in type alias declarations (#19410 by @WhoamiI00)

<!-- prettier-ignore -->
```ts
// Input
type A =
  /* 1 */ | /* 2 */ (
    /* 3 */ | /* 4 */ {
        key: string;
      }
  );

// Prettier stable
type A /* 2 */ = /* 1 */ /* 3 */ /* 4 */ {
  key: string;
};

// Prettier main
type A = /* 1 */ /* 2 */ /* 3 */ /* 4 */ {
  key: string;
};
```
