#### Don't break union type when it can fit (#18827 by @fisker)

<!-- prettier-ignore -->
```tsx
// Input
type Browser = "chromium" | "webkit" | "firefox" | "chromium-msedge" | "chromium-chrome";

// Prettier stable
type Browser =
  | "chromium"
  | "webkit"
  | "firefox"
  | "chromium-msedge"
  | "chromium-chrome";

// Prettier main
type Browser =
  "chromium" | "webkit" | "firefox" | "chromium-msedge" | "chromium-chrome";
```
