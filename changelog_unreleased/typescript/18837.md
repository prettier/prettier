#### Prevent moving comment in class properties (#18837 by @fisker)

<!-- prettier-ignore -->
```tsx
// Input
class foo {
  bar () /* bar */;
  baz /* baz */ ();
}

// Prettier stable
class foo {
  bar /* bar */();
  baz /* baz */();
}

// Prettier main
class foo {
  bar(); /* bar */
  baz /* baz */();
}
```
