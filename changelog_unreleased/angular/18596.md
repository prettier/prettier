#### Support spread element in array and object literals (#18596, #18636 by @prettier)

<!-- prettier-ignore -->
```html
<!-- Input -->
<component
  [a]="[   ...a]"
  [b]="{...bar  }"
  [c]="call(  ...baz)"
/>

<!-- Prettier stable -->
<component [a]="[   ...a]" [b]="{...bar  }" [c]="call(  ...baz)" />

<!-- Prettier main -->
<component [a]="[...a]" [b]="{ ...bar }" [c]="call(...baz)" />
```
