#### [HIGHLIGHT] Support support for spread elements (#18596, #18636 by @prettier)

This feature added in Angular v21.1.

<!-- prettier-ignore -->
```html
<!-- Input -->
<component
  [a]="[   ...a]"
  [b]="{...bar  }"
  [c]="call(  ...baz)"
/>

<!-- Prettier stable -->
<component [a]="[   ...a]" [b]="{...bar  }" [c]="call(  ...baz)" />

<!-- Prettier main -->
<component [a]="[...a]" [b]="{ ...bar }" [c]="call(...baz)" />
```
