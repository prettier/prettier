#### Support spread element in array and object literals (#18596 by @prettier)

<!-- prettier-ignore -->
```html
<!-- Input -->
<component
  [a]="[   ...a]"
  [b]="{...bar  }"
/>

<!-- Prettier stable -->
<component [a]="[   ...a]" [b]="{...bar  }" />

<!-- Prettier main -->
<component [a]="[...a]" [b]="{ ...bar }" />
```
