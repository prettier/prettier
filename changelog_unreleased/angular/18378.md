#### Don't print attribute values that are single template or string literals on new lines (#18378 by @ravindUwU)

<!-- prettier-ignore -->
```html
<!-- Input -->
<component
  [property]="`
    template
    literal
  `"
/>

<!-- Prettier stable -->
<component
  [property]="
    `
    template
    literal
  `
  "
/>

<!-- Prettier main -->
<component
  [property]="`
    template
    literal
  `"
/>
```
