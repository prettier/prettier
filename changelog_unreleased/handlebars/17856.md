#### Preserve `else if` syntax for custom helpers (#17856 by @kovsu)

<!-- prettier-ignore -->
```hbs
{{!-- Input --}}
{{#animated-if this.foo}}
foo content
{{else if (this.bar)}}
bar content
{{/animated-if}}

{{!-- Prettier stable --}}
{{#animated-if this.foo}}
  foo content
{{else}}{{#if (this.bar)}}
    bar content
  {{/if}}{{/animated-if}}

{{!-- Prettier main --}}
{{#animated-if this.foo}}
  foo content
{{else if (this.bar)}}
  bar content
{{/animated-if}}
```
