#### Support absolute path as plugin in config file (#15666 by @fisker)

```js
// prettier.config.cjs
module.exports = {
  plugins: [
    // posix style
    '/path/to/plugin.js',
    // Windows style
    'D:\\\\path\\to\\plugin.js',
    // Use `require.resolve`
    require.resolve('my-awesome-prettier-plugin'),
  ],
};
```
