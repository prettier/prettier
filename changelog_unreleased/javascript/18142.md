#### Remove redundant parentheses in "return statement" (#18142 by @fisker)

<!-- prettier-ignore -->
```jsx
// Input
function sequenceExpressionInside() {
  return ( // Reason for a
    a, b
  );
}

// Prettier stable
function sequenceExpressionInside() {
  return (
    // Reason for a
    (a, b)
  );
}

function sequenceExpressionInside() {
  return (
    // Reason for a
    a, b
  );
```
