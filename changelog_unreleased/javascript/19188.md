#### Remove extra spaces in `for` statement heads (#19188 by @Poliklot)

```jsx
// Input
for (initialize;;) {}
for (; condition;) {}

// Prettier stable
for (initialize; ; ) {}
for (; condition; ) {}
for (; ; update) {}

// Prettier main
for (initialize;;) {}
for (; condition;) {}
for (;; update) {}
```
