#### Remove indention in logical expression in `Boolean()` call (#18087 by @kovsu)

Reduce diff when changing a condition to an opposite value.

<!-- prettier-ignore -->
```js
// Input
const foo = Boolean(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);
const bar = !(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);

// Prettier stable
const foo = Boolean(
  a_long_long_long_long_condition ||
    a_long_long_long_long_condition ||
    a_long_long_long_long_condition,
);
const bar = !(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);

// Prettier main
const foo = Boolean(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition,
);
const bar = !(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);
```
