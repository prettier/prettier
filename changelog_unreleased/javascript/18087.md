#### Don't indent single argument in binary expressions (#18087 by @kovsu)

<!-- prettier-ignore -->
```js
// Input
Boolean(
  a_long_long_condition |
  a_long_long_condition |
  a_long_long_condition |
  a_long_long_condition |
  a_long_long_condition,
);

// Prettier stable
Boolean(
  a_long_long_condition |
    a_long_long_condition |
    a_long_long_condition |
    a_long_long_condition |
    a_long_long_condition,
);

// Prettier main
Boolean(
  a_long_long_condition |
  a_long_long_condition |
  a_long_long_condition |
  a_long_long_condition |
  a_long_long_condition,
);
```
