#### Do not indent a single logical expression argument (#18087 by @kovsu)

<!-- prettier-ignore -->
```js
// Input
const foo = Boolean(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);
const bar = !(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);

// Prettier stable
const foo = Boolean(
  a_long_long_long_long_condition ||
    a_long_long_long_long_condition ||
    a_long_long_long_long_condition,
);
const bar = !(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);

// Prettier main
const foo = Boolean(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition,
);
const bar = !(
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition ||
  a_long_long_long_long_condition
);
```
