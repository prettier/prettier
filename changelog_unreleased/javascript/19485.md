#### Keep simple top-level JSX on one line (#19485 by @macayu17)

<!-- prettier-ignore -->
```jsx
// Input
<a n:foo="bar"> {value} <b><c /></b></a>
<div>{a}{b}</div>

// Prettier stable
<a n:foo="bar">
  {" "}
  {value}{" "}
  <b>
    <c />
  </b>
</a>;
<div>
  {a}
  {b}
</div>;

// Prettier main
<a n:foo="bar"> {value} <b><c /></b></a>;
<div>{a}{b}</div>;
```
