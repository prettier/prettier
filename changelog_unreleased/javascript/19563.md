#### Avoid extra parentheses for nullish coalescing in JSX ternaries (#19563 by @sjh9714)

<!-- prettier-ignore -->
```jsx
// Input
const value = condition ? (
  <Example /> // comment
) : (
  "alpha" ?? "bravo"
);

// Prettier stable
const value = condition ? (
  <Example /> // comment
) : (
  ("alpha" ?? "bravo")
);

// Prettier main
const value = condition ? (
  <Example /> // comment
) : (
  "alpha" ?? "bravo"
);
```
