#### Avoid extra parentheses around nullish coalescing in JSX-mode ternaries (#19533 by @barry166)

<!-- prettier-ignore -->
```jsx
// Input
const value = condition ? (
  <Example /> // comment
) : (
  "alpha" ?? "bravo"
);

// Prettier stable
const value = condition ? (
  <Example /> // comment
) : (
  ("alpha" ?? "bravo")
);

// Prettier main
const value = condition ? (
  <Example /> // comment
) : (
  "alpha" ?? "bravo"
);
```
