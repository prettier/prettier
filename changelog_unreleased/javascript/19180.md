#### Fix multiline JSDoc type cast alignment in assignments (#19180 by @kovsu)

<!-- prettier-ignore -->
```jsx
// Input
const foo = /**
 * @type {{
 *   bar: string;
 * }}
 */ (baz);

// Prettier stable
const foo = /**
 * @type {{
 *   bar: string;
 * }}
 */ (baz);

// Prettier main
const foo =
  /**
   * @type {{
   *   bar: string;
   * }}
   */ (baz);
```
