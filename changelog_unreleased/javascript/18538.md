#### Keep iife function comments in parentheses (#18538 by @fisker)

<!-- prettier-ignore -->
```jsx
// Input
const a = (
  /**
   * @param {number} foo
   * @return {number}
   */
  function (foo) {
    return foo + 1;
  }
)(1);

// Prettier stable
const a = /**
 * @param {number} foo
 * @return {number}
 */
(function (foo) {
  return foo + 1;
})(1);

// Prettier main
const a = (
  /**
   * @param {number} foo
   * @return {number}
   */
  function (foo) {
    return foo + 1;
  }
)(1);
```
