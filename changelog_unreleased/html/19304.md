#### Keep an inline comment attached to the preceding node (#19304 by @kovsu)

<!-- prettier-ignore -->
```html
<!-- Input -->
<template>
  <div>{{ x }}<!-- comment --></div>
</template>

<!-- Prettier stable -->
<template>
  <div>
    {{ x
    }}<!-- comment -->
  </div>
</template>

<!-- Prettier main -->
<template>
  <div>{{ x }}<!-- comment --></div>
</template>
```
